package Leetcode;
 import java.util.*;
// Balanced Brackets
public class Z08 {
      public static boolean isBalenced(String s) {
    	  Stack<Character> stack=new Stack<Character>();
    	  for(char ch:s.toCharArray()) {
    		  switch (ch) {
			case '(': case '[': case '{':
				stack.push(ch);
				break;
			case ')':
				if(stack.isEmpty() || stack.pop() !='(')
					return false;
				break;
			case ']':
				if(stack.isEmpty() || stack.pop() !='[')
					return false;
				break;
			case '}':
				if(stack.isEmpty() || stack.pop() !='{')
					return false;
				break;
			}
    	  }
		return stack.isEmpty();	
	}
      public static void main(String[] args) {
		String s="){}}{()[]"; 
		
		System.out.println(isBalenced(s) + "\n" + isBalenced("()[{{}}]"));
	}      
}
public class ValidParenthesesArray {
    public static boolean isValid(String s) {
        // Use a char array as the stack
        char[] stack = new char[s.length()];
        int top = -1; // stack pointer

        for (char c : s.toCharArray()) {
            // Push opening brackets
            if (c == '(' || c == '{' || c == '[') {
                stack[++top] = c;
            } else {
                // If stack is empty, invalid
                if (top == -1) return false;

                char open = stack[top--]; // pop
                // Check matching pairs
                if ((c == ')' && open != '(') ||
                    (c == '}' && open != '{') ||
                    (c == ']' && open != '[')) {
                    return false;
                }
            }
        }
        // Valid only if stack is empty at the end
        return top == -1;
    }

    public static void main(String[] args) {
        System.out.println(isValid("()"));        // true
        System.out.println(isValid("([{}])"));    // true
        System.out.println(isValid("[(])"));      // false
        System.out.println(isValid("{[()]}"));    // true
        System.out.println(isValid("((("));       // false
    }
}
